\usepackage{titlesec}

  {\normalfont\bfseries\fontsize{16}{18}\selectfont}

  {1em}
  {}

  {0pt}
  {0pt}
  {24pt}

  {\normalfont\bfseries\fontsize{16}{18}\selectfont}

  {1em}
  {}

\titlespacing*{\section}
  {0pt}
  {24pt}
  {12pt}

  {1em}
  {}

  {12pt}

\newcounter{itemheading}[subsubsection]
\renewcommand{\theitemheading}{\thesubsubsection.\alph{itemheading}}
\newcommand{\itemheading}[1]{%
  \par\addvspace{12pt}%
  \refstepcounter{itemheading}%
  \noindent%
  {\fontsize{11}{13}\selectfont\bfseries\itshape\underline{\theitemheading. #1}}%
  \par\nobreak\vspace{6pt}%
}
