% ==================================
% Modern Table Template (Breakable)
% ==================================

% HOW TO USE THIS TEMPLATE:
% 1. This template uses the 'xltabular' package, which combines 'tabularx' (flexible columns)
%    and 'longtable' (tables that span multiple pages).
% 2. DO NOT wrap this in a \begin{table}[H] ... \end{table} environment! 'xltabular' handles 
%    page breaking and floats itself.
% 3. The \caption and \label MUST be placed inside the \begin{xltabular} environment.
% 4. You can define what appears at the top/bottom of the table on every page using:
%    - \endfirsthead: Header for the very first page of the table.
%    - \endhead: Header for all subsequent pages.
%    - \endfoot: Footer for all pages except the last one.
%    - \endlastfoot: Footer for the very last page of the table.
% 5. Remove the Comments and Use the table as per your need

% Column Definitions:
% |l| means left aligned with a border
% |c| means centered
% |X| means a dynamic width column that automatically wraps text to fit the page width

% Enclose in curly braces to limit the scope of \setModernTableStyle

{

% Applies the colors and spacing to match the template design
\setModernTableStyle 

\begin{xltabular}{\textwidth}{|l|X|c|}
    \hiderowcolors
    
    % The caption and label must be inside the table, followed by \\ 
    \caption{Description of the modern table}
    \label{tab:modern_example} \\
    \hline
    
    % --- 1. FIRST PAGE HEADER ---
    % Header Row: Use \tableHeaderStyle for the colored background and bold text
    \tableHeaderStyle Column 1 & Column 2 (Wraps text) & Column 3 \\ \thickhline
    \endfirsthead % End of first page header definition
    
    % --- 2. SUBSEQUENT PAGES HEADER ---
    % This appears at the top of the table on page 2, page 3, etc.
    \tableHeaderStyle Column 1 & Column 2 (Wraps text) & Column 3 \\ \thickhline
    \endhead % End of subsequent pages header definition
    \showrowcolors
    
    % --- 3. FOOTER FOR ALL PAGES EXCEPT THE LAST ---
    % This appears at the bottom of the table on page 1, page 2, etc. (uncomment to use if allowed)
    % \multicolumn{3}{r}{\textit{Continued on next page...}} \\
    \endfoot % End of footer definition
    
    % --- 4. LAST PAGE FOOTER ---
    % This appears at the very bottom of the table on the final page
    \endlastfoot % End of last page footer definition

    % --- 5. DATA ROWS ---
    % Add your table data here. The table will automatically break to the next page when full.
    Row 1 Data & Long text will wrap nicely to the next line here. & Yes \\ \hline
    Row 2 Data & 200 & No  \\ \hline
    Row 3 Data & 300 & Yes \\ \hline
    Row 4 Data & This table can now automatically break across multiple pages if it becomes too long. & Yes \\ \hline
    Row 5 Data & Just keep adding rows as you normally would. & Yes \\ \hline
    
\end{xltabular}
}
